14. Write a C++ program for Default Constructors and parameterized Constructor.
//default constructor
#include<iostream>
using namespace std;
class A
{
 private:
 int a;
 public:
 A()
 {
 cout << "enter a value:";
 cin >> a;
 cout << "a = " << a;
 }
};
int main()
{
 A obj;
}
//parameterized constructor
#include<iostream>
using namespace std;
class A
{
 private:
 int a;
 public:
 A(int a)
 {
 cout << "a = " << a;
 }
};
int main()
{
 int a;
 cout << "enter a value:";
 cin >> a;
 A obj(a);
}
