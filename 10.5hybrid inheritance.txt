Hybrid inheritance
#include<iostream>
using namespace std;
class length
{
 protected:
 int l;
 public:
 length()
 {
 cout << "Enter length:";
 cin >> l;
 }
};
class breadth
{
 protected:
 int b;
 public:
 breadth()
 {
 cout << "Enter breadth:";
 cin >> b;
 }
};
class area : public length, public breadth
{
 protected:
 int a;
 public:

 {
 a = l*b;
 cout << "area = " << a << endl;
 }
